41  public class IbatorConfigurationParser {
42      private Properties properties;

44      public IbatorConfigurationParser(Properties properties) {
45          super();
46          if (properties == null) {
47              this.properties = System.getProperties();
48          } else {
49              this.properties = properties;
50          }
51      }

53      public Configuration parseIbatorConfiguration(Element rootNode)
54              throws XMLParserException {
55  
56          Configuration configuration = new Configuration();
57  
58          NodeList nodeList = rootNode.getChildNodes();
59          for (int i = 0; i < nodeList.getLength(); i++) {
60              Node childNode = nodeList.item(i);
61  
62              if (childNode.getNodeType() != Node.ELEMENT_NODE) {
63                  continue;
64              }
65  
66              if ("properties".equals(childNode.getNodeName())) { //$NON-NLS-1$
67                  parseProperties(configuration, childNode);
68              } else if ("classPathEntry".equals(childNode.getNodeName())) { //$NON-NLS-1$
69                  parseClassPathEntry(configuration, childNode);
70              } else if ("ibatorContext".equals(childNode.getNodeName())) { //$NON-NLS-1$
71                  parseIbatorContext(configuration, childNode);
72              }
73          }
74  
75          return configuration;
76      }

78      private void parseProperties(Configuration configuration, Node node)
79              throws XMLParserException {
80          Properties attributes = parseAttributes(node);
81          String resource = attributes.getProperty("resource"); //$NON-NLS-1$
82          String url = attributes.getProperty("url"); //$NON-NLS-1$
83  
84          if (!stringHasValue(resource)
85                  && !stringHasValue(url)) {
86              throw new XMLParserException(getString("RuntimeError.14")); //$NON-NLS-1$
87          }
88  
89          if (stringHasValue(resource)
90                  && stringHasValue(url)) {
91              throw new XMLParserException(getString("RuntimeError.14")); //$NON-NLS-1$
92          }
93  
94          URL resourceUrl;
95  
96          try {
97              if (stringHasValue(resource)) {
98                  resourceUrl = ObjectFactory.getResource(resource);
99                  if (resourceUrl == null) {
100                     throw new XMLParserException(getString(
101                             "RuntimeError.15", resource)); //$NON-NLS-1$
102                 }
103             } else {
104                 resourceUrl = new URL(url);
105             }
106 
107             InputStream inputStream = resourceUrl.openConnection()
108                     .getInputStream();
109 
110             properties.load(inputStream);
111             inputStream.close();
112         } catch (IOException e) {
113             if (stringHasValue(resource)) {
114                 throw new XMLParserException(getString(
115                         "RuntimeError.16", resource)); //$NON-NLS-1$
116             } else {
117                 throw new XMLParserException(getString(
118                         "RuntimeError.17", url)); //$NON-NLS-1$
119             }
120         }
121     }

123     private void parseIbatorContext(Configuration configuration, Node node) {
124 
125         Properties attributes = parseAttributes(node);
126         String defaultModelType = attributes.getProperty("defaultModelType"); //$NON-NLS-1$
127         String targetRuntime = attributes.getProperty("targetRuntime"); //$NON-NLS-1$
128         String introspectedColumnImpl = attributes
129                 .getProperty("introspectedColumnImpl"); //$NON-NLS-1$
130         String id = attributes.getProperty("id"); //$NON-NLS-1$
131 
132         ModelType mt = defaultModelType == null ? null : ModelType
133                 .getModelType(defaultModelType);
134 
135         Context context = new Context(mt);
136         context.setId(id);
137         if (stringHasValue(introspectedColumnImpl)) {
138             context.setIntrospectedColumnImpl(introspectedColumnImpl);
139         }
140         if (stringHasValue(targetRuntime)) {
141             context.setTargetRuntime(targetRuntime);
142         }
143 
144         configuration.addContext(context);
145 
146         NodeList nodeList = node.getChildNodes();
147         for (int i = 0; i < nodeList.getLength(); i++) {
148             Node childNode = nodeList.item(i);
149 
150             if (childNode.getNodeType() != Node.ELEMENT_NODE) {
151                 continue;
152             }
153 
154             if ("property".equals(childNode.getNodeName())) { //$NON-NLS-1$
155                 parseProperty(context, childNode);
156             } else if ("ibatorPlugin".equals(childNode.getNodeName())) { //$NON-NLS-1$
157                 parseIbatorPlugin(context, childNode);
158             } else if ("commentGenerator".equals(childNode.getNodeName())) { //$NON-NLS-1$
159                 parseCommentGenerator(context, childNode);
160             } else if ("jdbcConnection".equals(childNode.getNodeName())) { //$NON-NLS-1$
161                 parseJdbcConnection(context, childNode);
162             } else if ("javaModelGenerator".equals(childNode.getNodeName())) { //$NON-NLS-1$
163                 parseJavaModelGenerator(context, childNode);
164             } else if ("javaTypeResolver".equals(childNode.getNodeName())) { //$NON-NLS-1$
165                 parseJavaTypeResolver(context, childNode);
166             } else if ("sqlMapGenerator".equals(childNode.getNodeName())) { //$NON-NLS-1$
167                 parseSqlMapGenerator(context, childNode);
168             } else if ("daoGenerator".equals(childNode.getNodeName())) { //$NON-NLS-1$
169                 parseDaoGenerator(context, childNode);
170             } else if ("table".equals(childNode.getNodeName())) { //$NON-NLS-1$
171                 parseTable(context, childNode);
172             }
173         }
174     }

202     private void parseTable(Context context, Node node) {
203         TableConfiguration tc = new TableConfiguration(context);
204         context.addTableConfiguration(tc);
205 
206         Properties attributes = parseAttributes(node);
207         String catalog = attributes.getProperty("catalog"); //$NON-NLS-1$
208         String schema = attributes.getProperty("schema"); //$NON-NLS-1$
209         String tableName = attributes.getProperty("tableName"); //$NON-NLS-1$
210         String domainObjectName = attributes.getProperty("domainObjectName"); //$NON-NLS-1$
211         String alias = attributes.getProperty("alias"); //$NON-NLS-1$
212         String enableInsert = attributes.getProperty("enableInsert"); //$NON-NLS-1$
213         String enableSelectByPrimaryKey = attributes
214                 .getProperty("enableSelectByPrimaryKey"); //$NON-NLS-1$
215         String enableSelectByExample = attributes
216                 .getProperty("enableSelectByExample"); //$NON-NLS-1$
217         String enableUpdateByPrimaryKey = attributes
218                 .getProperty("enableUpdateByPrimaryKey"); //$NON-NLS-1$
219         String enableDeleteByPrimaryKey = attributes
220                 .getProperty("enableDeleteByPrimaryKey"); //$NON-NLS-1$
221         String enableDeleteByExample = attributes
222                 .getProperty("enableDeleteByExample"); //$NON-NLS-1$
223         String enableCountByExample = attributes
224                 .getProperty("enableCountByExample"); //$NON-NLS-1$
225         String enableUpdateByExample = attributes
226                 .getProperty("enableUpdateByExample"); //$NON-NLS-1$
227         String selectByPrimaryKeyQueryId = attributes
228                 .getProperty("selectByPrimaryKeyQueryId"); //$NON-NLS-1$
229         String selectByExampleQueryId = attributes
230                 .getProperty("selectByExampleQueryId"); //$NON-NLS-1$
231         String modelType = attributes.getProperty("modelType"); //$NON-NLS-1$
232         String escapeWildcards = attributes.getProperty("escapeWildcards"); //$NON-NLS-1$
233         String delimitIdentifiers = attributes
234                 .getProperty("delimitIdentifiers"); //$NON-NLS-1$
235         String delimitAllColumns = attributes.getProperty("delimitAllColumns"); //$NON-NLS-1$
236 
237         if (stringHasValue(catalog)) {
238             tc.setCatalog(catalog);
239         }
240 
241         if (stringHasValue(schema)) {
242             tc.setSchema(schema);
243         }
244 
245         if (stringHasValue(tableName)) {
246             tc.setTableName(tableName);
247         }
248 
249         if (stringHasValue(domainObjectName)) {
250             tc.setDomainObjectName(domainObjectName);
251         }
252 
253         if (stringHasValue(alias)) {
254             tc.setAlias(alias);
255         }
256 
257         if (stringHasValue(enableInsert)) {
258             tc.setInsertStatementEnabled(isTrue(enableInsert));
259         }
260 
261         if (stringHasValue(enableSelectByPrimaryKey)) {
262             tc.setSelectByPrimaryKeyStatementEnabled(isTrue(enableSelectByPrimaryKey));
263         }
264 
265         if (stringHasValue(enableSelectByExample)) {
266             tc.setSelectByExampleStatementEnabled(isTrue(enableSelectByExample));
267         }
268 
269         if (stringHasValue(enableUpdateByPrimaryKey)) {
270             tc.setUpdateByPrimaryKeyStatementEnabled(isTrue(enableUpdateByPrimaryKey));
271         }
272 
273         if (stringHasValue(enableDeleteByPrimaryKey)) {
274             tc.setDeleteByPrimaryKeyStatementEnabled(isTrue(enableDeleteByPrimaryKey));
275         }
276 
277         if (stringHasValue(enableDeleteByExample)) {
278             tc.setDeleteByExampleStatementEnabled(isTrue(enableDeleteByExample));
279         }
280 
281         if (stringHasValue(enableCountByExample)) {
282             tc.setCountByExampleStatementEnabled(isTrue(enableCountByExample));
283         }
284 
285         if (stringHasValue(enableUpdateByExample)) {
286             tc.setUpdateByExampleStatementEnabled(isTrue(enableUpdateByExample));
287         }
288 
289         if (stringHasValue(selectByPrimaryKeyQueryId)) {
290             tc.setSelectByPrimaryKeyQueryId(selectByPrimaryKeyQueryId);
291         }
292 
293         if (stringHasValue(selectByExampleQueryId)) {
294             tc.setSelectByExampleQueryId(selectByExampleQueryId);
295         }
296 
297         if (stringHasValue(modelType)) {
298             tc.setConfiguredModelType(modelType);
299         }
300 
301         if (stringHasValue(escapeWildcards)) {
302             tc.setWildcardEscapingEnabled(isTrue(escapeWildcards));
303         }
304 
305         if (stringHasValue(delimitIdentifiers)) {
306             tc.setDelimitIdentifiers(isTrue(delimitIdentifiers));
307         }
308 
309         if (stringHasValue(delimitAllColumns)) {
310             tc.setAllColumnDelimitingEnabled(isTrue(delimitAllColumns));
311         }
312 
313         NodeList nodeList = node.getChildNodes();
314         for (int i = 0; i < nodeList.getLength(); i++) {
315             Node childNode = nodeList.item(i);
316 
317             if (childNode.getNodeType() != Node.ELEMENT_NODE) {
318                 continue;
319             }
320 
321             if ("property".equals(childNode.getNodeName())) { //$NON-NLS-1$
322                 parseProperty(tc, childNode);
323             } else if ("columnOverride".equals(childNode.getNodeName())) { //$NON-NLS-1$
324                 parseColumnOverride(tc, childNode);
325             } else if ("ignoreColumn".equals(childNode.getNodeName())) { //$NON-NLS-1$
326                 parseIgnoreColumn(tc, childNode);
327             } else if ("generatedKey".equals(childNode.getNodeName())) { //$NON-NLS-1$
328                 parseGeneratedKey(tc, childNode);
329             } else if ("columnRenamingRule".equals(childNode.getNodeName())) { //$NON-NLS-1$
330                 parseColumnRenamingRule(tc, childNode);
331             }
332         }

335     private void parseColumnOverride(TableConfiguration tc, Node node) {
336         Properties attributes = parseAttributes(node);
337         String column = attributes.getProperty("column"); //$NON-NLS-1$
338         String property = attributes.getProperty("property"); //$NON-NLS-1$
339         String javaType = attributes.getProperty("javaType"); //$NON-NLS-1$
340         String jdbcType = attributes.getProperty("jdbcType"); //$NON-NLS-1$
341         String typeHandler = attributes.getProperty("typeHandler"); //$NON-NLS-1$
342         String delimitedColumnName = attributes
343                 .getProperty("delimitedColumnName"); //$NON-NLS-1$
344 
345         ColumnOverride co = new ColumnOverride(column);
346 
347         if (stringHasValue(property)) {
348             co.setJavaProperty(property);
349         }
350 
351         if (stringHasValue(javaType)) {
352             co.setJavaType(javaType);
353         }
354 
355         if (stringHasValue(jdbcType)) {
356             co.setJdbcType(jdbcType);
357         }
358 
359         if (stringHasValue(typeHandler)) {
360             co.setTypeHandler(typeHandler);
361         }
362 
363         if (stringHasValue(delimitedColumnName)) {
364             co.setColumnNameDelimited(isTrue(delimitedColumnName));
365         }
366 
367         NodeList nodeList = node.getChildNodes();
368         for (int i = 0; i < nodeList.getLength(); i++) {
369             Node childNode = nodeList.item(i);
370 
371             if (childNode.getNodeType() != Node.ELEMENT_NODE) {
372                 continue;
373             }
374 
375             if ("property".equals(childNode.getNodeName())) { //$NON-NLS-1$
376                 parseProperty(co, childNode);
377             }
378         }
379 
380         tc.addColumnOverride(co);
381     }

588     private Properties parseAttributes(Node node) {
589         Properties attributes = new Properties();
590         NamedNodeMap nnm = node.getAttributes();
591         for (int i = 0; i < nnm.getLength(); i++) {
592             Node attribute = nnm.item(i);
593             String value = parsePropertyTokens(attribute.getNodeValue());
594             attributes.put(attribute.getNodeName(), value);
595         }
596 
597         return attributes;
598     }
599 
600     private String parsePropertyTokens(String string) {
601         final String OPEN = "${"; //$NON-NLS-1$
602         final String CLOSE = "}"; //$NON-NLS-1$
603 
604         String newString = string;
605         if (newString != null) {
606             int start = newString.indexOf(OPEN);
607             int end = newString.indexOf(CLOSE);
608 
609             while (start > -1 && end > start) {
610                 String prepend = newString.substring(0, start);
611                 String append = newString.substring(end + CLOSE.length());
612                 String propName = newString.substring(start + OPEN.length(),
613                         end);
614                 String propValue = properties.getProperty(propName);
615                 if (propValue != null) {
616                     newString = prepend + propValue + append;
617                 }
618 
619                 start = newString.indexOf(OPEN, end);
620                 end = newString.indexOf(CLOSE, end);
621             }
622         }
623 
624         return newString;
625     }
