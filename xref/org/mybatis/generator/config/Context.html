133     /**
134      * This method does a simple validate, it makes sure that all required
135      * fields have been filled in. It does not do any more complex operations
136      * such as validating that database tables exist or validating that named
137      * columns exist
138      */
139     public void validate(List<String> errors) {
140         if (!stringHasValue(id)) {
141             errors.add(getString("ValidationError.16")); //$NON-NLS-1$
142         }
143 
144         if (jdbcConnectionConfiguration == null) {
145             errors.add(getString("ValidationError.10", id)); //$NON-NLS-1$
146         } else {
147             jdbcConnectionConfiguration.validate(errors);
148         }
149 
150         if (javaModelGeneratorConfiguration == null) {
151             errors.add(getString("ValidationError.8", id)); //$NON-NLS-1$
152         } else {
153             javaModelGeneratorConfiguration.validate(errors, id);
154         }
155 
156         if (extjsGeneratorConfiguration != null) {
157             extjsGeneratorConfiguration.validate(errors, id);
158         }
159 
160         if (javaClientGeneratorConfiguration != null) {
161             javaClientGeneratorConfiguration.validate(errors, id);
162         }
163 
164         IntrospectedTable it = null;
165         try {
166             it = ObjectFactory.createIntrospectedTableForValidation(this);
167         } catch (Exception e) {
168             errors.add(getString("ValidationError.25", id)); //$NON-NLS-1$
169         }
170         
171         if (it != null && it.requiresXMLGenerator()) {
172             if (sqlMapGeneratorConfiguration == null) {
173                 errors.add(getString("ValidationError.9", id)); //$NON-NLS-1$
174             } else {
175                 sqlMapGeneratorConfiguration.validate(errors, id);
176             }
177         }
178 
179         if (tableConfigurations.size() == 0) {
180             errors.add(getString("ValidationError.3", id)); //$NON-NLS-1$
181         } else {
182             for (int i = 0; i < tableConfigurations.size(); i++) {
183                 TableConfiguration tc = tableConfigurations.get(i);
184 
185                 tc.validate(errors, i);
186             }
187         }
188 
189         for (PluginConfiguration pluginConfiguration : pluginConfigurations) {
190             pluginConfiguration.validate(errors, id);
191         }
192     }

236     /**
237      * Builds an XmlElement representation of this context. Note that the XML
238      * may not necessarily validate if the context is invalid. Call the
239      * <code>validate</code> method to check validity of this context.
240      * 
241      * @return the XML representation of this context
242      */
243     public XmlElement toXmlElement() {
244         XmlElement xmlElement = new XmlElement("context"); //$NON-NLS-1$
245         
246         xmlElement.addAttribute(new Attribute("id", id)); //$NON-NLS-1$
247         
248         if (defaultModelType != ModelType.CONDITIONAL) {
249             xmlElement.addAttribute(new Attribute(
250                     "defaultModelType", defaultModelType.getModelType())); //$NON-NLS-1$
251         }
252 
253         if (stringHasValue(introspectedColumnImpl)) {
254             xmlElement.addAttribute(new Attribute(
255                     "introspectedColumnImpl", introspectedColumnImpl)); //$NON-NLS-1$
256         }
257 
258         if (stringHasValue(targetRuntime)) {
259             xmlElement.addAttribute(new Attribute(
260                     "targetRuntime", targetRuntime)); //$NON-NLS-1$
261         }
262 
263         addPropertyXmlElements(xmlElement);
264         
265         for (PluginConfiguration pluginConfiguration : pluginConfigurations) {
266             xmlElement.addElement(pluginConfiguration.toXmlElement());
267         }
268 
269         if (commentGeneratorConfiguration != null) {
270             xmlElement.addElement(commentGeneratorConfiguration.toXmlElement());
271         }
272 
273         if (jdbcConnectionConfiguration != null) {
274             xmlElement.addElement(jdbcConnectionConfiguration.toXmlElement());
275         }
276 
277         if (javaTypeResolverConfiguration != null) {
278             xmlElement.addElement(javaTypeResolverConfiguration.toXmlElement());
279         }
280 
281         if (javaModelGeneratorConfiguration != null) {
282             xmlElement.addElement(javaModelGeneratorConfiguration
283                     .toXmlElement());
284         }
285 
286         if (extjsGeneratorConfiguration != null) {
287             xmlElement.addElement(extjsGeneratorConfiguration
288                     .toXmlElement());
289         }
290 
291         if (sqlMapGeneratorConfiguration != null) {
292             xmlElement.addElement(sqlMapGeneratorConfiguration.toXmlElement());
293         }
294 
295         if (javaClientGeneratorConfiguration != null) {
296             xmlElement.addElement(javaClientGeneratorConfiguration.toXmlElement());
297         }
298 
299         for (TableConfiguration tableConfiguration : tableConfigurations) {
300             xmlElement.addElement(tableConfiguration.toXmlElement());
301         }
302 
303         return xmlElement;
304     }

393     // methods related to code generation.
394     //
395     // Methods should be called in this order:
396     //
397     // 1. getIntrospectionSteps()
398     // 2. introspectTables()
399     // 3. getGenerationSteps()
400     // 4. generateFiles()
401     //
402 
403     private List<IntrospectedTable> introspectedTables;
404 
405     public int getIntrospectionSteps() {
406         int steps = 0;
407 
408         steps++; // connect to database
409 
410         // for each table:
411         //
412         // 1. Create introspected table implementation
413 
414         steps += tableConfigurations.size() * 1;
415 
416         return steps;
417     }

419     /**
420      * Introspect tables based on the configuration specified in the
421      * constructor. This method is long running.
422      * 
423      * @param callback
424      *            a progress callback if progress information is desired, or
425      *            <code>null</code>
426      * @param warnings
427      *            any warning generated from this method will be added to the
428      *            List. Warnings are always Strings.
429      * @param fullyQualifiedTableNames
430      *            a set of table names to generate. The elements of the set must
431      *            be Strings that exactly match what's specified in the
432      *            configuration. For example, if table name = "foo" and schema =
433      *            "bar", then the fully qualified table name is "foo.bar". If
434      *            the Set is null or empty, then all tables in the configuration
435      *            will be used for code generation.
436      * 
437      * @throws SQLException
438      *             if some error arises while introspecting the specified
439      *             database tables.
440      * @throws InterruptedException
441      *             if the progress callback reports a cancel
442      */
443     public void introspectTables(ProgressCallback callback,
444             List<String> warnings, Set<String> fullyQualifiedTableNames)
445             throws SQLException, InterruptedException {
446 
447         introspectedTables = new ArrayList<IntrospectedTable>();
448         JavaTypeResolver javaTypeResolver = ObjectFactory
449                 .createJavaTypeResolver(this, warnings);
450 
451         Connection connection = null;
452 
453         try {
454             callback.startTask(getString("Progress.0")); //$NON-NLS-1$
455             connection = getConnection();
456 
457             DatabaseIntrospector databaseIntrospector = new DatabaseIntrospector(
458                     this, connection.getMetaData(), javaTypeResolver, warnings);
459 
460             for (TableConfiguration tc : tableConfigurations) {
461                 String tableName = composeFullyQualifiedTableName(tc.getCatalog(), tc
462                                 .getSchema(), tc.getTableName(), '.');
463 
464                 if (fullyQualifiedTableNames != null
465                         && fullyQualifiedTableNames.size() > 0) {
466                     if (!fullyQualifiedTableNames.contains(tableName)) {
467                         continue;
468                     }
469                 }
470 
471                 if (!tc.areAnyStatementsEnabled()) {
472                     warnings.add(getString("Warning.0", tableName)); //$NON-NLS-1$
473                     continue;
474                 }
475 
476                 callback.startTask(getString("Progress.1", tableName)); //$NON-NLS-1$
477                 List<IntrospectedTable> tables = databaseIntrospector
478                         .introspectTables(tc);
479 
480                 if (tables != null) {
481                     introspectedTables.addAll(tables);
482                 }
483 
484                 callback.checkCancel();
485             }
486         } finally {
487             closeConnection(connection);
488         }
489     }
490 
491     public int getGenerationSteps() {
492         int steps = 0;
493 
494         if (introspectedTables != null) {
495             for (IntrospectedTable introspectedTable : introspectedTables) {
496                 steps += introspectedTable.getGenerationSteps();
497             }
498         }
499 
500         return steps;
501     }

503     public void generateFiles(ProgressCallback callback,
504             List<GeneratedJavaFile> generatedJavaFiles,
505             List<GeneratedXmlFile> generatedXmlFiles, List<String> warnings)
506             throws InterruptedException {
507 
508         pluginAggregator = new PluginAggregator();
509         for (PluginConfiguration pluginConfiguration : pluginConfigurations) {
510             Plugin plugin = ObjectFactory.createPlugin(this,
511                     pluginConfiguration);
512             if (plugin.validate(warnings)) {
513                 pluginAggregator.addPlugin(plugin);
514             } else {
515                 warnings.add(getString("Warning.24", //$NON-NLS-1$
516                         pluginConfiguration.getConfigurationType(), id));
517             }
518         }
519 
520         if (introspectedTables != null) {
521             for (IntrospectedTable introspectedTable : introspectedTables) {
522                 callback.checkCancel();
523 
524                 introspectedTable.initialize();
525                 introspectedTable.calculateGenerators(warnings, callback);
526                 generatedJavaFiles.addAll(introspectedTable
527                         .getGeneratedJavaFiles());
528                 generatedXmlFiles.addAll(introspectedTable
529                         .getGeneratedXmlFiles());
530 
531                 generatedJavaFiles.addAll(pluginAggregator
532                         .contextGenerateAdditionalJavaFiles(introspectedTable));
533                 generatedXmlFiles.addAll(pluginAggregator
534                         .contextGenerateAdditionalXmlFiles(introspectedTable));
535             }
536         }
537 
538         generatedJavaFiles.addAll(pluginAggregator
539                 .contextGenerateAdditionalJavaFiles());
540         generatedXmlFiles.addAll(pluginAggregator
541                 .contextGenerateAdditionalXmlFiles());
542     }
