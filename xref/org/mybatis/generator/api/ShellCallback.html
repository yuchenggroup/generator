16  package org.mybatis.generator.api;
17  
18  import java.io.File;
19  
20  import org.mybatis.generator.exception.ShellException;
21  
22  /**
23   * This interface defines methods that a shell should support to enable
24   * the generator
25   * to work. A "shell" is defined as the execution environment (i.e. an
26   * Eclipse plugin, and Ant task, a NetBeans plugin, etc.)
27   * 
28   * The default ShellCallback that is very low function and does
29   * not support the merging of Java files. The default shell callback is 
30   * appropriate for use in well controlled environments where no changes
31   * made to generated Java files.
32   * 
33   * @author Jeff Butler
34   */
35  public interface ShellCallback {
36      /**
37       * This method is called to ask the shell to resolve a
38       * project/package combination into a directory on the file system.
39       * This method is called repeatedly (once for each generated file), so it
40       * would be wise for an implementing class to cache results.
41       * 
42       * The returned <code>java.io.File</code> object:
43       * <ul>
44       * <li>Must be a directory</li>
45       * <li>Must exist</li>
46       * </ul>
47       * 
48       * The default shell callback interprets both values as directories and
49       * simply concatenates the two values to generate the default directory.
50       * 
51       * @param targetProject
52       * @param targetPackage
53       * @return the directory (must exist)
54       * @throws ShellException
55       *             if the project/package cannot be resolved into a directory on
56       *             the file system. In this case, the generator will not save the file
57       *             it is currently working on. The generator will add the exception
58       *             message to the list of warnings automatically.
59       */
60      File getDirectory(String targetProject, String targetPackage)
61              throws ShellException;
62  
63      /**
64       * This method is called if a newly generated Java file would
65       * overwrite an existing file. This method should return the merged source
66       * (formatted). The generator will write the merged source as-is to the file
67       * system.
68       * 
69       * A merge typically follows these steps:
70       * <ol>
71       * <li>Delete any methods/fields in the existing file that have the
72       * specified JavaDoc tag</li>
73       * <li>Add any new super interfaces from the new file into the existing file
74       * </li>
75       * <li>Make sure that the existing file's super class matches the new file</li>
76       * <li>Make sure that the existing file is of the same type as the existing
77       * file (either interface or class)</li>
78       * <li>Add any new imports from the new file into the existing file</li>
79       * <li>Add all methods and fields from the new file into the existing file</li>
80       * <li>Format the resulting source string</li>
81       * </ol>
82       * 
83       * This method is called only if you return <code>true</code> from
84       * <code>isMergeSupported()</code>.
85       * 
86       * @param newFileSource
87       *            the source of the newly generated Java file
88       * @param existingFileFullPath
89       *            the fully qualified path name of the existing Java file
90       * @param javadocTags
91       *            the JavaDoc tags that denotes which methods and fields in the
92       *            old file to delete (if the Java element has any of these tags,
93       *            the element is eligible for merge)
94       * @param fileEncoding
95       *            the file encoding for reading existing Java files.  Can be null,
96       *            in which case the platform default encoding will be used.
97       * @return the merged source, properly formatted. The source will be saved
98       *         exactly as returned from this method.
99       * @throws ShellException
100      *             if the file cannot be merged for some reason. If this
101      *             exception is thrown, nothing will be saved and the
102      *             existing file will remain undisturbed. The generator will add the
103      *             exception message to the list of warnings automatically.
104      */
105     String mergeJavaFile(String newFileSource, String existingFileFullPath,
106             String[] javadocTags, String fileEncoding) throws ShellException;
107 
108     /**
109      * After all files are saved to the file system, this method is called
110      * once for each unique project that was affected by the generation
111      * run. This method is useful if your IDE needs to be informed that file
112      * system objects have been created or updated. If you are running 
113      * outside of an IDE, your implementation need not do anything in this
114      * method.
115      * 
116      * @param project
117      *            the project to be refreshed
118      */
119     void refreshProject(String project);
120 
121     /**
122      * Return true if the callback supports Java merging, otherwise false.
123      * The <code>mergeJavaFile()</code> method will be called only if this
124      * method returns <code>true</code>.
125      * 
126      * @return a boolean specifying whether Java merge is supported or not
127      */
128     boolean isMergeSupported();
129 
130     /**
131      * Return true if the generator should overwrite an existing file if one exists.
132      * This method will be called only if <code>isMergeSupported()</code>
133      * returns <code>false</code> and a file exists that would be overwritten by
134      * a generated file. If you return <code>true</code>, then we will log a
135      * warning specifying what file was overwritten.
136      * 
137      * @return true if you want to overwrite existing files
138      */
139     boolean isOverwriteEnabled();
140 }
