56      /**
57       * This object is used to hold information related to the table itself, not
58       * the columns in the table.
59       * 
60       * @param introspectedCatalog
61       *            the actual catalog of the table as returned from
62       *            DatabaseMetaData. This value should only be set if the user
63       *            configured a catalog. Otherwise the DatabaseMetaData is
64       *            reporting some database default that we don't want in the
65       *            generated code.
66       * 
67       * @param introspectedSchema
68       *            the actual schema of the table as returned from
69       *            DatabaseMetaData. This value should only be set if the user
70       *            configured a schema. Otherwise the DatabaseMetaData is
71       *            reporting some database default that we don't want in the
72       *            generated code.
73       * 
74       * @param introspectedTableName
75       *            the actual table name as returned from DatabaseMetaData
76       * 
77       * @param domainObjectName
78       *            the configured domain object name for this table. If nothing
79       *            is configured, we'll build the domain object named based on
80       *            the tableName or runtimeTableName.
81       * 
82       * @param alias
83       *            a configured alias for the table. This alias will be added to
84       *            the table name in the SQL
85       * 
86       * @param ignoreQualifiersAtRuntime
87       *            if true, then the catalog and schema qualifiers will be
88       *            ignored when composing fully qualified names in the generated
89       *            SQL. This is used, for example, when the user needs to specify
90       *            a specific schema for generating code but does not want the
91       *            schema in the generated SQL
92       * 
93       * @param runtimeCatalog
94       *            this is used to "rename" the catalog in the generated SQL.
95       *            This is useful, for example, when generating code against one
96       *            catalog that should run with a different catalog.
97       * 
98       * @param runtimeSchema
99       *            this is used to "rename" the schema in the generated SQL. This
100      *            is useful, for example, when generating code against one
101      *            schema that should run with a different schema.
102      * 
103      * @param runtimeTableName
104      *            this is used to "rename" the table in the generated SQL. This
105      *            is useful, for example, when generating code to run with an
106      *            Oracle synonym. The user would have to specify the actual
107      *            table name and schema for generation, but would want to use
108      *            the synonym name in the generated SQL
109      * 
110      * @param delimitIdentifiers
111      *            if true, then the table identifiers will be delimited at
112      *            runtime. The delimiter characters are obtained from the
113      *            Context.
114      */
115     public FullyQualifiedTable(String introspectedCatalog,
116             String introspectedSchema, String introspectedTableName,
117             String domainObjectName, String alias,
118             boolean ignoreQualifiersAtRuntime, String runtimeCatalog,
119             String runtimeSchema, String runtimeTableName,
120             boolean delimitIdentifiers, Context context) {
121         super();
122         this.introspectedCatalog = introspectedCatalog;
123         this.introspectedSchema = introspectedSchema;
124         this.introspectedTableName = introspectedTableName;
125         this.ignoreQualifiersAtRuntime = ignoreQualifiersAtRuntime;
126         this.runtimeCatalog = runtimeCatalog;
127         this.runtimeSchema = runtimeSchema;
128         this.runtimeTableName = runtimeTableName;
129         
130         if (stringHasValue(domainObjectName)) {
131             int index = domainObjectName.lastIndexOf('.');
132             if (index == -1) {
133                 this.domainObjectName = domainObjectName;
134             } else {
135                 this.domainObjectName = domainObjectName.substring(index + 1);
136                 this.domainObjectSubPackage = domainObjectName.substring(0, index);
137             }
138         }
139 
140         if (alias == null) {
141             this.alias = null;
142         } else {
143             this.alias = alias.trim();
144         }
145 
146         beginningDelimiter = delimitIdentifiers ? context
147                 .getBeginningDelimiter() : ""; //$NON-NLS-1$
148         endingDelimiter = delimitIdentifiers ? context.getEndingDelimiter()
149                 : ""; //$NON-NLS-1$
150     }

164     /**
165      * @return
166      */
167     public String getFullyQualifiedTableNameAtRuntime() {
168         StringBuilder localCatalog = new StringBuilder();
169         if (!ignoreQualifiersAtRuntime) {
170             if (stringHasValue(runtimeCatalog)) {
171                 localCatalog.append(runtimeCatalog);
172             } else if (stringHasValue(introspectedCatalog)) {
173                 localCatalog.append(introspectedCatalog);
174             }
175         }
176         if (localCatalog.length() > 0) {
177             addDelimiters(localCatalog);
178         }
179 
180         StringBuilder localSchema = new StringBuilder();
181         if (!ignoreQualifiersAtRuntime) {
182             if (stringHasValue(runtimeSchema)) {
183                 localSchema.append(runtimeSchema);
184             } else if (stringHasValue(introspectedSchema)) {
185                 localSchema.append(introspectedSchema);
186             }
187         }
188         if (localSchema.length() > 0) {
189             addDelimiters(localSchema);
190         }
191 
192         StringBuilder localTableName = new StringBuilder();
193         if (stringHasValue(runtimeTableName)) {
194             localTableName.append(runtimeTableName);
195         } else {
196             localTableName.append(introspectedTableName);
197         }
198         addDelimiters(localTableName);
199 
200         return composeFullyQualifiedTableName(localCatalog
201                 .toString(), localSchema.toString(), localTableName.toString(),
202                 '.');
203     }

205     /**
206      * @return
207      */
208     public String getAliasedFullyQualifiedTableNameAtRuntime() {
209         StringBuilder sb = new StringBuilder();
210 
211         sb.append(getFullyQualifiedTableNameAtRuntime());
212 
213         if (stringHasValue(alias)) {
214             sb.append(' ');
215             sb.append(alias);
216         }
217 
218         return sb.toString();
219     }

221     /**
222      * This method returns a string that is the fully qualified table name, with
223      * underscores as the separator.
224      * 
225      * @return the namespace
226      */
227     public String getIbatis2SqlMapNamespace() {
228         String localCatalog = stringHasValue(runtimeCatalog) ? runtimeCatalog
229                 : introspectedCatalog;
230         String localSchema = stringHasValue(runtimeSchema) ? runtimeSchema
231                 : introspectedSchema;
232         String localTable = stringHasValue(runtimeTableName) ? runtimeTableName
233                 : introspectedTableName;
234 
235         return composeFullyQualifiedTableName(
236                         ignoreQualifiersAtRuntime ? null : localCatalog,
237                         ignoreQualifiersAtRuntime ? null : localSchema,
238                         localTable, '_');
239     }

241     public String getDomainObjectName() {
242         if (stringHasValue(domainObjectName)) {
243             return domainObjectName;
244         } else if (stringHasValue(runtimeTableName)) {
245             return getCamelCaseString(runtimeTableName, true);
246         } else {
247             return getCamelCaseString(introspectedTableName, true);
248         }
249     }

251     @Override
252     public boolean equals(Object obj) {
253         if (this == obj) {
254             return true;
255         }
256 
257         if (!(obj instanceof FullyQualifiedTable)) {
258             return false;
259         }
260 
261         FullyQualifiedTable other = (FullyQualifiedTable) obj;
262 
263         return areEqual(this.introspectedTableName,
264                 other.introspectedTableName)
265                 && areEqual(this.introspectedCatalog,
266                         other.introspectedCatalog)
267                 && areEqual(this.introspectedSchema,
268                         other.introspectedSchema);
269     }

292     /**
293      * Calculates a Java package fragment based on the table catalog and schema.
294      * If qualifiers are ignored, then this method will return an empty string
295      * 
296      * @return the subpackage for this table
297      */
298     public String getSubPackage(boolean isSubPackagesEnabled) {
299         StringBuilder sb = new StringBuilder();
300         if (!ignoreQualifiersAtRuntime && isSubPackagesEnabled) {
301             if (stringHasValue(runtimeCatalog)) {
302                 sb.append('.');
303                 sb.append(runtimeCatalog.toLowerCase());
304             } else if (stringHasValue(introspectedCatalog)) {
305                 sb.append('.');
306                 sb.append(introspectedCatalog.toLowerCase());
307             }
308 
309             if (stringHasValue(runtimeSchema)) {
310                 sb.append('.');
311                 sb.append(runtimeSchema.toLowerCase());
312             } else if (stringHasValue(introspectedSchema)) {
313                 sb.append('.');
314                 sb.append(introspectedSchema.toLowerCase());
315             }
316         }
317         
318         if (stringHasValue(domainObjectSubPackage)) {
319             sb.append('.');
320             sb.append(domainObjectSubPackage);
321         }
322 
323         // TODO - strip characters that are not valid in package names
324         return sb.toString();
325     }
326 
327     private void addDelimiters(StringBuilder sb) {
328         if (stringHasValue(beginningDelimiter)) {
329             sb.insert(0, beginningDelimiter);
330         }
331 
332         if (stringHasValue(endingDelimiter)) {
333             sb.append(endingDelimiter);
334         }
335     }
