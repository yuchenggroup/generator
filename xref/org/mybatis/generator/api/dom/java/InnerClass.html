26  /**
27   * This class encapsulates the idea of an inner class - it has methods that make
28   * it easy to generate inner classes.
29   * 
30   * @author Jeff Butler
31   */
32  public class InnerClass extends JavaElement {
33      private List<Field> fields;
34  
35      private List<InnerClass> innerClasses;
36  
37      private List<InnerEnum> innerEnums;
38  
39      private FullyQualifiedJavaType superClass;
40  
41      private FullyQualifiedJavaType type;
42  
43      private Set<FullyQualifiedJavaType> superInterfaceTypes;
44  
45      private List<Method> methods;
46  
47      private boolean isAbstract;
48      
49      private List<InitializationBlock> initializationBlocks;
50  
51      /**
52       *  
53       */
54      public InnerClass(FullyQualifiedJavaType type) {
55          super();
56          this.type = type;
57          fields = new ArrayList<Field>();
58          innerClasses = new ArrayList<InnerClass>();
59          innerEnums = new ArrayList<InnerEnum>();
60          superInterfaceTypes = new HashSet<FullyQualifiedJavaType>();
61          methods = new ArrayList<Method>();
62          initializationBlocks = new ArrayList<InitializationBlock>();
63      }
64  
65      public InnerClass(String typeName) {
66          this(new FullyQualifiedJavaType(typeName));
67      }

71       */
72      public List<Field> getFields() {
73          return fields;
74      }
75

77          fields.add(field);
78      }

84          return superClass;
85      }

87      /**
88       * @param superClass
89       *            The superClass to set.
90       */
91      public void setSuperClass(FullyQualifiedJavaType superClass) {
92          this.superClass = superClass;
93      }

95      public void setSuperClass(String superClassType) {
96          this.superClass = new FullyQualifiedJavaType(superClassType);
97      }

126     public String getFormattedContent(int indentLevel) {
127         StringBuilder sb = new StringBuilder();
128 
129         addFormattedJavadoc(sb, indentLevel);
130         addFormattedAnnotations(sb, indentLevel);
131 
132         OutputUtilities.javaIndent(sb, indentLevel);
133         sb.append(getVisibility().getValue());
134 
135         if (isAbstract()) {
136             sb.append("abstract "); //$NON-NLS-1$
137         }
138 
139         if (isStatic()) {
140             sb.append("static "); //$NON-NLS-1$
141         }
142 
143         if (isFinal()) {
144             sb.append("final "); //$NON-NLS-1$
145         }
146 
147         sb.append("class "); //$NON-NLS-1$
148         sb.append(getType().getShortName());
149 
150         if (superClass != null) {
151             sb.append(" extends "); //$NON-NLS-1$
152             sb.append(superClass.getShortName());
153         }
154 
155         if (superInterfaceTypes.size() > 0) {
156             sb.append(" implements "); //$NON-NLS-1$
157 
158             boolean comma = false;
159             for (FullyQualifiedJavaType fqjt : superInterfaceTypes) {
160                 if (comma) {
161                     sb.append(", "); //$NON-NLS-1$
162                 } else {
163                     comma = true;
164                 }
165 
166                 sb.append(fqjt.getShortName());
167             }
168         }
169 
170         sb.append(" {"); //$NON-NLS-1$
171         indentLevel++;
172         
173         Iterator<Field> fldIter = fields.iterator();
174         while (fldIter.hasNext()) {
175             OutputUtilities.newLine(sb);
176             Field field = fldIter.next();
177             sb.append(field.getFormattedContent(indentLevel));
178             if (fldIter.hasNext()) {
179                 OutputUtilities.newLine(sb);
180             }
181         }
182 
183         if (initializationBlocks.size() > 0) {
184             OutputUtilities.newLine(sb);
185         }
186 
187         Iterator<InitializationBlock> blkIter = initializationBlocks.iterator();
188         while (blkIter.hasNext()) {
189             OutputUtilities.newLine(sb);
190             InitializationBlock initializationBlock = blkIter.next();
191             sb.append(initializationBlock.getFormattedContent(indentLevel));
192             if (blkIter.hasNext()) {
193                 OutputUtilities.newLine(sb);
194             }
195         }
196 
197         if (methods.size() > 0) {
198             OutputUtilities.newLine(sb);
199         }
200 
201         Iterator<Method> mtdIter = methods.iterator();
202         while (mtdIter.hasNext()) {
203             OutputUtilities.newLine(sb);
204             Method method = mtdIter.next();
205             sb.append(method.getFormattedContent(indentLevel, false));
206             if (mtdIter.hasNext()) {
207                 OutputUtilities.newLine(sb);
208             }
209         }
210 
211         if (innerClasses.size() > 0) {
212             OutputUtilities.newLine(sb);
213         }
214         Iterator<InnerClass> icIter = innerClasses.iterator();
215         while (icIter.hasNext()) {
216             OutputUtilities.newLine(sb);
217             InnerClass innerClass = icIter.next();
218             sb.append(innerClass.getFormattedContent(indentLevel));
219             if (icIter.hasNext()) {
220                 OutputUtilities.newLine(sb);
221             }
222         }
223 
224         if (innerEnums.size() > 0) {
225             OutputUtilities.newLine(sb);
226         }
227 
228         Iterator<InnerEnum> ieIter = innerEnums.iterator();
229         while (ieIter.hasNext()) {
230             OutputUtilities.newLine(sb);
231             InnerEnum innerEnum = ieIter.next();
232             sb.append(innerEnum.getFormattedContent(indentLevel));
233             if (ieIter.hasNext()) {
234                 OutputUtilities.newLine(sb);
235             }
236         }
237 
238         indentLevel--;
239         OutputUtilities.newLine(sb);
240         OutputUtilities.javaIndent(sb, indentLevel);
241         sb.append('}');
242 
243         return sb.toString();
244     }
