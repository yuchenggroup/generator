122     public void addConfigurationProperties(Properties properties) {
123         this.properties.putAll(properties);
124         forceBigDecimals = StringUtility
125                 .isTrue(properties
126                         .getProperty(PropertyRegistry.TYPE_RESOLVER_FORCE_BIG_DECIMALS));
127     }
128 
129     public FullyQualifiedJavaType calculateJavaType(
130             IntrospectedColumn introspectedColumn) {
131         FullyQualifiedJavaType answer;
132         JdbcTypeInformation jdbcTypeInformation = typeMap
133                 .get(introspectedColumn.getJdbcType());
134 
135         if (jdbcTypeInformation == null) {
136             switch (introspectedColumn.getJdbcType()) {
137             case Types.DECIMAL:
138             case Types.NUMERIC:
139                 if (introspectedColumn.getScale() > 0
140                         || introspectedColumn.getLength() > 18
141                         || forceBigDecimals) {
142                     answer = new FullyQualifiedJavaType(BigDecimal.class
143                             .getName());
144                 } else if (introspectedColumn.getLength() > 9) {
145                     answer = new FullyQualifiedJavaType(Long.class.getName());
146                 } else if (introspectedColumn.getLength() > 4) {
147                     answer = new FullyQualifiedJavaType(Integer.class.getName());
148                 } else {
149                     answer = new FullyQualifiedJavaType(Short.class.getName());
150                 }
151                 break;
152 
153             default:
154                 answer = null;
155                 break;
156             }
157         } else {
158             answer = jdbcTypeInformation.getFullyQualifiedJavaType();
159         }
160 
161         return answer;
162     }
163 
164     public String calculateJdbcTypeName(IntrospectedColumn introspectedColumn) {
165         String answer;
166         JdbcTypeInformation jdbcTypeInformation = typeMap
167                 .get(introspectedColumn.getJdbcType());
168 
169         if (jdbcTypeInformation == null) {
170             switch (introspectedColumn.getJdbcType()) {
171             case Types.DECIMAL:
172                 answer = "DECIMAL"; //$NON-NLS-1$
173                 break;
174             case Types.NUMERIC:
175                 answer = "NUMERIC"; //$NON-NLS-1$
176                 break;
177             default:
178                 answer = null;
179                 break;
180             }
181         } else {
182             answer = jdbcTypeInformation.getJdbcTypeName();
183         }
184 
185         return answer;
186     }
